24  *
25  * <p>
26  * This class provides a {@link java.security.KeyStore} which stores :

28  * <li> The private key ({@link java.security.PrivateKey}).</li>
29  * <li>

32  * </li>

35  * password.
36  * </p>

84  /**

87  * @return {@link java.security.KeyPair} object.

94  static KeyPair getKSKeyPair() throws IOException, KeyStoreException, CertificateException, NoSuchAlgorithmException, UnrecoverableEntryException {
95  FileInputStream fis = new FileInputStream(KEY_STORE_NAME);
96 
97  KeyStore keyStore = KeyStore.getInstance(KEY_STORE_TYPE);
98  keyStore.load(fis, KEY_STORE_PASSWORD.toCharArray());
99 
100  Key key = keyStore.getKey(KEY_STORE_ALIAS, KEY_STORE_PASSWORD.toCharArray());
101  if(key instanceof PrivateKey) {
102  PublicKey publicKey = keyStore.getCertificate(KEY_STORE_ALIAS).getPublicKey();
103  return new KeyPair(publicKey, (PrivateKey) key);
104  }
105  return null;
106  }

116  private static String generateKSPassword() {
117  Preferences preferences = Preferences.userNodeForPackage(KeyStoreManager.class);
118  String password = preferences.get("Password", null);
119  if(password == null) {
120  SecureRandom random = new SecureRandom();
121  StringBuilder stringBuilder = new StringBuilder(new BigInteger(36, 0, random).toString(Character.MAX_RADIX));
122  while (stringBuilder.length() > 6) {
123  stringBuilder.deleteCharAt(random.nextInt(stringBuilder.length()));
124  }
125  for (int i = 0; i < stringBuilder.length(); i++) {
126  char ch = stringBuilder.charAt(i);
127  if (Character.isLetter(ch) && Character.isLowerCase(ch) && random.nextFloat() < 0.5) {
128  stringBuilder.setCharAt(i, Character.toUpperCase(ch));
129  }
130  }
131  password = stringBuilder.toString();
132  preferences.put("Password", password);
133  }
134  return password;
135  }
