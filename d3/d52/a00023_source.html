118  private void broadcastData(byte[] data) throws IOException {
119  Enumeration<NetworkInterface> nis = NetworkInterface.getNetworkInterfaces();
120  while (nis.hasMoreElements()) {
121  NetworkInterface ni = nis.nextElement();
122  if (!ni.isLoopback()) {
123  List<InterfaceAddress> ias = ni.getInterfaceAddresses();
124  for(InterfaceAddress addr: ias) {
125  InetAddress broadcastIA = addr.getBroadcast();
126  if(broadcastIA != null) {
127  DatagramSocket datagramSocket = new DatagramSocket();
128  datagramSocket.setBroadcast(true);
129  DatagramPacket datagramPacket = new DatagramPacket(data, data.length,
130  broadcastIA, BROADCAST_PORT);
131  datagramSocket.send(datagramPacket);
132  datagramSocket.close();
133  }
134  }
135  }
