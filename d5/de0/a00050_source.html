51 /**
52  * Utility for the end-to-end security module.
53  *
54  * <p>
55  * This class provides the following features:
56  * <ul>
57  * <li>
58  * <i>256-bit Elliptic Curve</i> private and
59  * public key generation.
60  * </li>
61  * <li><i>AES 256-bit</i> plaintext encryption.</li>
62  * <li><i>AES 256-bit</i> cipher text decryption.</li>
63  * <li>Self-signed certificate generation.</li>
64  * </ul>
65  *
66  * The private-public key pair and the self-signed
67  * certificate is generated by the constructor
68  * during instantiation and is stored in a keystore
69  * ({@link java.security.KeyStore}) maintained by the
70  * {@link KeyStoreManager}. If a <code>KeyStore</code>
71  * already exists, the key pair and the certificate
72  * gets loaded directly. The encryption and
73  * decryption mechanism is provided by the
74  * {@link #encryptString(String)} and
75  * {@link #decryptString(String)} methods respectively.
76  * </p>
77  *
78  * @see project.pc.security.KeyStoreManager
79  * @see java.security.KeyStore
80  * @see java.security.KeyPair
81  * @see project.pc.security.EKEProvider#encryptString(String)
82  * @see project.pc.security.EKEProvider#decryptString(String)
83  */

139  /**
140  * Constructor.
141  *
142  * <p>
143  * Initializes this <code>EKEProvider</code>.
144  * Generates a secret key using the <i>SHA-256
145  * message digest</i> of the pairing key generated
146  * by {@link EKEProvider#getPairingKey()} and the
147  * public key of the client.
148  * </p>
149  *
150  * @param pairingKey Pairing key used for client-server
151  * authentication.
152  * @param clientPublicKey Public key of the client.
153  * @see project.pc.security.KeyStoreManager
154  */
155  public EKEProvider(String pairingKey, byte[] clientPublicKey) {
156  MessageDigest msgDigest;
157  try {
158  msgDigest = MessageDigest.getInstance(MESSAGE_DIGEST, SECURITY_PROVIDER);
159  mIV = msgDigest.digest(pairingKey.getBytes());
160  KeyPair keyPair = KeyStoreManager.getKSKeyPair();
161  if (keyPair != null) {
162  X509EncodedKeySpec pubKeySpec = new X509EncodedKeySpec(Base64.decodeBase64(clientPublicKey));
163  KeyFactory keyFactory = KeyFactory.getInstance(KEY_GENERATION_ALGORITHM, SECURITY_PROVIDER);
164  mSecretKey = generateSharedSecret(keyPair.getPrivate(), keyFactory.generatePublic(pubKeySpec));
165  }
166  } catch (NoSuchAlgorithmException | NoSuchProviderException
167  | CertificateException | KeyStoreException | UnrecoverableEntryException
168  | IOException | InvalidKeySpecException e) {
169  e.printStackTrace();
170  }
171  }

173  private static String generatePairingKey() {
174  SecureRandom random = new SecureRandom();
175  StringBuilder stringBuilder = new StringBuilder(new BigInteger(36, 0, random).toString(Character.MAX_RADIX));
176  while(stringBuilder.length() > 6) {
177  stringBuilder.deleteCharAt(random.nextInt(stringBuilder.length()));
178  }
179  for (int i = 0; i < stringBuilder.length(); i++) {
180  char ch = stringBuilder.charAt(i);
181  if (Character.isLetter(ch) && Character.isLowerCase(ch) && random.nextFloat() < 0.5) {
182  stringBuilder.setCharAt(i, Character.toUpperCase(ch));
183  }
184  }
185  return stringBuilder.toString();
186  }

286  private Certificate genSelfSignedCert(PublicKey publicKey, PrivateKey privateKey)
287  throws OperatorCreationException, CertificateException {
288  X500Principal issuer = new X500Principal("CN=127.0.0.1");
289  BigInteger serial = BigInteger.valueOf(System.currentTimeMillis());
290  X509v3CertificateBuilder certGen =
291  new JcaX509v3CertificateBuilder(issuer, serial, NOT_BEFORE, NOT_AFTER,
292  issuer, publicKey);
293  X509CertificateHolder certHolder = certGen.build(new JcaContentSignerBuilder(SIGNATURE_ALGORITHM)
294  .setProvider(SECURITY_PROVIDER)
295  .build(privateKey));
296  return new JcaX509CertificateConverter()
297  .setProvider(SECURITY_PROVIDER)
298  .getCertificate(certHolder);
299  }
