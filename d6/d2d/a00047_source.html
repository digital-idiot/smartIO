18 /**
19  * Implementation of the {@link java.lang.Runnable} for the thread that is
20  * responsible for communication with each client.
21  *
22  * <p>
23  * The actions performed by this thread are :
24  * <ul>
25  * <li>Authenticate the client.</li>
26  * <li>According to the result
27  * <ul>
28  * <li>
29  * If the client is not authentic, close the connection
30  * and terminate.
31  * </li>
32  * <li>
33  * Otherwise, confirm the connection and start
34  * communication with the client.
35  * </li>
36  * </ul>
37  * </li>
38  * <li>
39  * Examine the data received from client and transfer it to
40  * a <i>controller</i> ({@link project.pc.device.MouseController}
41  * or {@link project.pc.device.KeyboardController}) to execute the
42  * input operations.
43  * </li>
44  * </ul>
45  * </p>
46  * <p>
47  * Client authentication is done by comparing the pairing key received
48  * from the client with the one displayed on the pairing key dialog
49  * window.
50  * </p>
51  * <p>
52  * It receives data from client and wraps it in a {@link DataWrapper}
53  * object. It, then, invokes the {@link DataWrapper#getOperationType()}
54  * method to examine the type of operation requested by the client and
55  * accordingly transfers the control to either
56  * {@link project.pc.device.MouseController} or
57  * {@link project.pc.device.KeyboardController}.
58  * </p>
59  * <p>
60  * This thread is started by the {@link Server#listen()} upon accepting
61  * every client connection. It gets terminated implicitly after
62  * unsuccessful authentication of the client. Explicitly, it is terminated
63  * by the {@link NetworkManager} ({@link NetworkManager#stopServer()} and
64  * {@link NetworkManager#disconnect(InetAddress)}) by invoking the
65  * {@link #setStopFlag()} method at the time of server stop or
66  * disconnection of this client A reference of this thread gets stored
67  * in the {@link NetworkState} after successful authentication of the
68  * client.
69  * </p>
70  *
71  * @see java.lang.Runnable
72  * @see project.pc.net.DataWrapper
73  * @see project.pc.net.DataWrapper#getOperationType()
74  * @see project.pc.device.MouseController
75  * @see project.pc.device.KeyboardController
76  * @see project.pc.net.NetworkManager#stopServer()
77  * @see project.pc.net.NetworkManager#disconnect(InetAddress)
78  */

242  private void receiveData() throws InterruptedException {
243  try {
244  if(mClientSocket.isClosed()) return;
245  BufferedReader in = new BufferedReader(new InputStreamReader(mClientSocket.getInputStream()));
246  if(!in.ready()) return;
247  String readData = mEKEProvider.decryptString(in.readLine());
248  DataWrapper dataWrapper = new Gson().fromJson(readData, DataWrapper.class);
249  if(dataWrapper != null) {
250  switch (dataWrapper.getOperationType()) {
251  case "Stop":
252  setStopFlag();
253  break;
254 
255  case "Mouse_Move":
256  mMouseController.move(dataWrapper.getQuaternionObject(), dataWrapper.isInitQuat(), dataWrapper.getSensitivity());
257  break;
258 
259  case "Mouse_Touch":
260  mMouseController.move(dataWrapper.getX(), dataWrapper.getY(), dataWrapper.getSensitivity());
261  break;
262 
263  case "Mouse_Button":
264  mMouseController.doOperation(dataWrapper.getData());
265  break;
266 
267  case "Key":
268  mKeyboardController.doKeyOperation(dataWrapper.getData());
269  break;
270 
271  case "Special_Key":
272  mKeyboardController.doSpecialKeyOperation(dataWrapper.getData());
273  }
274  }
275  } catch (IOException e) {
276  e.printStackTrace();
277  }
278  }
