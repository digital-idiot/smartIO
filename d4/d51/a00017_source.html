220  List<String> selectedValues = mList.getSelectedValuesList();
221  try {
222  for (String values: selectedValues) {
223  values = values.substring(values.indexOf("(") + 1, values.length() - 1);
224  mManager.disconnect(InetAddress.getByName(values));
225  }
226  } catch (IOException | InterruptedException e) {
227  e.printStackTrace();
228  }
229  System.out.println("disconnect clicked");
230  }
231 
232  else if(event.getActionCommand().equals("on") && !mLastSelectedOption) {
233  mLastSelectedOption = true;
234  System.out.println("on clicked");
235  mThread = new Thread(mNetworkThread);
236  mThread.start();
237  }
238 
239  else if(event.getActionCommand().equals("off") && mLastSelectedOption) {
240  mLastSelectedOption = false;
241  System.out.println("off clicked");
242  try {
243  mManager.stopServer();
244  mThread = null;
245  } catch (Exception e) { e.printStackTrace(); }
246  }
247  else if(event.getActionCommand().equals("3d_clicked")) {
248  if(!mState.isNoClientConnected() && !mCube.isStarted())
249  mCube.showCube();
250  }
251  }
