12 /**
13  * Utility for performing mouse input operations.
14  *
15  * <p>
16  * It uses a {@link java.awt.Robot} object to generate native
17  * system input events. This differs from posting events to
18  * the AWT event queue or AWT components in that the events are
19  * generated in the platform's native input queue.
20  * </p>
21  * <p>
22  * The {@link #move(Quaternion, boolean, float)} and
23  * {@link #move(int, int, float)} methods are invoked for moving the
24  * mouse pointer on the basis of 3-D and 2-D inputs from the
25  * mobile device respectively. The {@link #move(Quaternion, boolean, float)}
26  * method, in turn, calls the
27  * {@link MouseMove#moveRelatively(Robot, int, int, float)} method.
28  * The mouse button inputs are performed by invoking the
29  * {@link #doOperation(String)} method.
30  * </p>
31  *
32  * @see java.awt.Robot
33  * @see project.pc.sensor.representation.Quaternion
34  * @see project.pc.device.MouseMove#moveRelatively(Robot, int, int, float)
35  */

66  public void move(Quaternion quaternion, boolean isInitQuat, float sensitivity) {
67  sQuaternion = quaternion;
68  if(isInitQuat) {
69  mSensorDataHandler = new SensorDataHandler(quaternion,sensitivity);
70  }
71  Cartesian2D cartesian2D = mSensorDataHandler.pointerUpdate(quaternion);
72  int x = MouseInfo.getPointerInfo().getLocation().x;
73  int y = MouseInfo.getPointerInfo().getLocation().y;
74  x += (int) cartesian2D.getX();
75  y += (int) cartesian2D.getY();
76  mRobot.mouseMove(x, y);
77  }
