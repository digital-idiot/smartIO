8 /**
9  * Stores the state information about the network.
10  *
11  * <p>
12  * It stores the information about the client connections with the help of
13  * two data structures :
14  * <ol>
15  * <li>
16  * A <i>Connection map</i> which is a <code>HashMap</code> from a
17  * client <code>Socket</code> to the {@link ServerThread} handling
18  * that client.
19  * </li>
20  * <li>
21  * An <i>Address map</i> which is a <code>HashMap</code> from the
22  * <code>InetAddress</code> of a client to the <code>Socket</code>.
23  * </li>
24  * </ol>
25  * The methods to manipulate these data structures are provided in the class.
26  * The {@link #add(ClientInfo, ServerThread)} and {@link #remove(ClientInfo)}
27  * methods are invoked by {@link ServerThread} once it successfully
28  * establishes and closes a client connection respectively.
29  * </p>
30  * <p>
31  * It also stores a {@link javax.swing.DefaultListModel} which is used by
32  * {@link project.pc.gui.MainWindow} to populate a list displaying the client connections
33  * in the GUI.
34  * </p>
35  *
36  * @see java.util.HashMap
37  * @see project.pc.net.ServerThread
38  * @see project.pc.gui.MainWindow
39  * @see javax.swing.DefaultListModel
40  */

121  String client = clientInfo.getClientInfo() + " (" + socket.getInetAddress().getHostAddress() + ")";
122  if(mListModel.contains(client)) mListModel.removeElement(client);
123  }
