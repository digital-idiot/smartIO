3 /**
4  * Class to represent a quaternion.
5  *
6  * <p>
7  * A quaternion <i>H</i> may be identified with R<sup>4</sup>,
8  * a four-dimensional vector space over the real numbers. The
9  * elements of this basis are customarily denoted as <i>1</i>,
10  * <i>i</i>, <i>j</i>, and <i>k</i>. Every element of <i>H</i>
11  * can be uniquely written as a linear combination of these
12  * basis elements, that is, as <i>a1 + bi + cj + dk</i>, where
13  * <i>a</i>, <i>b</i>, <i>c</i>, and <i>d</i> are real
14  * numbers.
15  * </p>
16  * <p>
17  * Quaternions allow for elegant descriptions of 3D rotations,
18  * interpolations as well as extrapolations and compared to
19  * Euler angles, they don't suffer from gimbal lock.
20  * Interpolations between two Quaternions are called SLERP
21  * (Spherical Linear Interpolation).
22  * </p>
23  * <p>
24  * If <i>a + bi + cj + dk</i> is any quaternion, then <i>a</i>
25  * is called its scalar part and <i>bi + cj + dk</i> is called
26  * its vector part. The scalar part of a quaternion is always
27  * real, and the vector part is always pure imaginary.
28  * </p>
29  * <p>
30  * Using the basis <i>1, i, j, k</i> of H it is possible to
31  * write <i>H</i> as a set of quadruples:
32  * </p>
33  * <p>
34  * &nbsp;&nbsp;&nbsp;&nbsp;
35  * <b><i>H = {(a,b,c,d) | a,b,c,d &isin; R}</i></b>.
36  * </p>
37  * <p>
38  * Then the basis elements are: <br/>
39  * <i>1 = (1,0,0,0)</i>, <br/>
40  * <i>i = (0,1,0,0)</i>, <br/>
41  * <i>j = (0,0,1,0)</i>, <br/>
42  * <i>k = (0,0,0,1)</i>. <br/>
43  * </p>
44  * <p>
45  * <b>Addition of two quaternions : </b><br/>
46  * <i>
47  * (a<sub>1</sub>,b<sub>1</sub>,c<sub>1</sub>,d<sub>1</sub>) +
48  * (a<sub>2</sub>,b<sub>2</sub>,c<sub>2</sub>,d<sub>2</sub>) =
49  * (a<sub>1</sub>+a<sub>2</sub>, b<sub>1</sub>+b<sub>2</sub>,
50  * c<sub>1</sub>+c<sub>2</sub>, d<sub>1</sub>+d<sub>2</sub>)
51  * </i>.
52  * </p>
53  * <p>
54  * <b>Multiplications of two quaternions : </b><br/>
55  * <i>
56  * (a<sub>1</sub>,b<sub>1</sub>,c<sub>1</sub>,d<sub>1</sub>)
57  * (a<sub>2</sub>,b<sub>2</sub>,c<sub>2</sub>,d<sub>2</sub>) =
58  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
59  * (a<sub>1</sub>a<sub>2</sub> - b<sub>1</sub>b<sub>2</sub> -
60  * c<sub>1</sub>c<sub>2</sub> - d<sub>1</sub>d<sub>2</sub>,
61  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
62  * a<sub>1</sub>b<sub>2</sub> + b<sub>1</sub>a<sub>2</sub> +
63  * c<sub>1</sub>d<sub>2</sub> - d<sub>1</sub>c<sub>2</sub>,
64  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
65  * a<sub>1</sub>c<sub>2</sub> - b<sub>1</sub>d<sub>2</sub> +
66  * c<sub>1</sub>a<sub>2</sub> + d<sub>1</sub>b<sub>2</sub>,
67  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
68  * a<sub>1</sub>d<sub>2</sub> + b<sub>1</sub>c<sub>2</sub> -
69  * c<sub>1</sub>b<sub>2</sub> + d<sub>1</sub>a<sub>2</sub>)
70  * </i>.
71  * </p>
72  * <p>
73  * <b>Normalization of quaternion</b> <i>(a,b,c,d)</i> <br/>
74  * &nbsp;&nbsp;&nbsp;&nbsp;
75  * = <i>(a/mag, b/mag, c/mag, d/mag)</i>, <br/>
76  * where, <br/> &nbsp;&nbsp;&nbsp;&nbsp;
77  * <i>mag =
78  * &radic;(a<sup>2</sup>+b<sup>2</sup>+c<sup>2</sup>+d<sup>2</sup>)</i>.
79  * </p>
80  * <p>
81  * The sensor data for 3-D mouse movement is sent from the
82  * mobile device is represented in the form of a
83  * <code>Quaternion</code>.
84  * </p>
85  *
86  * @see project.pc.sensor.representation.Vector4f
87  */

304  private void generateQuaternionFromMatrix() {
305 
306  float qx;
307  float qy;
308  float qz;
309  float qw;
310 
311  float[] mat = mMatrix.getMatrix();
312  int[] indices;
313 
314  if (this.mMatrix.size() == 16) {
315  if (this.mMatrix.isColumnMajor()) {
316  indices = MatrixF4x4.MAT_IND_COL16_3X3;
317  } else {
318  indices = MatrixF4x4.MAT_IND_ROW16_3X3;
319  }
320  } else {
321  if (this.mMatrix.isColumnMajor()) {
322  indices = MatrixF4x4.MAT_IND_COL9_3X3;
323  } else {
324  indices = MatrixF4x4.MAT_IND_ROW9_3X3;
325  }
326  }
327 
328  int m00 = indices[0];
329  int m01 = indices[1];
330  int m02 = indices[2];
331 
332  int m10 = indices[3];
333  int m11 = indices[4];
334  int m12 = indices[5];
335 
336  int m20 = indices[6];
337  int m21 = indices[7];
338  int m22 = indices[8];
339 
340  float tr = mat[m00] + mat[m11] + mat[m22];
341  if (tr > 0) {
342  float s = (float) Math.sqrt(tr + 1.0) * 2;
343  qw = 0.25f * s;
344  qx = (mat[m21] - mat[m12]) / s;
345  qy = (mat[m02] - mat[m20]) / s;
346  qz = (mat[m10] - mat[m01]) / s;
347  } else if ((mat[m00] > mat[m11]) & (mat[m00] > mat[m22])) {
348  float s = (float) Math.sqrt(1.0 + mat[m00] - mat[m11] - mat[m22]) * 2;
349  qw = (mat[m21] - mat[m12]) / s;
350  qx = 0.25f * s;
351  qy = (mat[m01] + mat[m10]) / s;
352  qz = (mat[m02] + mat[m20]) / s;
353  } else if (mat[m11] > mat[m22]) {
354  float s = (float) Math.sqrt(1.0 + mat[m11] - mat[m00] - mat[m22]) * 2;
355  qw = (mat[m02] - mat[m20]) / s;
356  qx = (mat[m01] + mat[m10]) / s;
357  qy = 0.25f * s;
358  qz = (mat[m12] + mat[m21]) / s;
359  } else {
360  float s = (float) Math.sqrt(1.0 + mat[m22] - mat[m00] - mat[m11]) * 2;
361  qw = (mat[m10] - mat[m01]) / s;
362  qx = (mat[m02] + mat[m20]) / s;
363  qy = (mat[m12] + mat[m21]) / s;
364  qz = 0.25f * s;
365  }
366 
367  setX(qx);
368  setY(qy);
369  setZ(qz);
370  setW(qw);
371  }

391  double heading = Math.toRadians(roll);
392  double attitude = Math.toRadians(pitch);
393  double bank = Math.toRadians(azimuth);

398  double s2 = Math.sin(attitude / 2);
399  double c3 = Math.cos(bank / 2);
400  double s3 = Math.sin(bank / 2);
401  double c1c2 = c1 * c2;
402  double s1s2 = s1 * s2;
403  setW((float) (c1c2 * c3 - s1s2 * s3));
404  setX((float) (c1c2 * s3 + s1s2 * c3));
405  setY((float) (s1 * c2 * c3 + c1 * s2 * s3));
406  setZ((float) (c1 * s2 * c3 - s1 * c2 * s3));
407 
408  mDirty = true;
409  }

443  /**
444  * Computes a linear interpolation between this <code>Quaternion</code> and the
445  * input <code>Quaternion</code>.
446  *
447  * @param input the <code>Quaternion</code> to be slerped with this.
448  * @param output the <code>Quaternion</code> to store the result.
449  * @param t the ratio between the two quaternions where 0 <= t <= 1.0 . <br/>
450  * <b><i>Note:</i></b> Increase value of <code>t</code> will bring
451  * rotation closer to the input <code>Quaternion</code>.
452  */
453  public void slerp(Quaternion input, Quaternion output, float t) {
454  // Calculate angle between them
455  Quaternion bufferQuat;
456  float cosHalftheta = this.dotProduct(input);
457 
458  if (cosHalftheta < 0) {
459  if(mTmpQuaternion == null) mTmpQuaternion = new Quaternion();
460  bufferQuat = mTmpQuaternion;
461  cosHalftheta = -cosHalftheta;
462  bufferQuat.mPoints[0] = (-input.mPoints[0]);
463  bufferQuat.mPoints[1] = (-input.mPoints[1]);
464  bufferQuat.mPoints[2] = (-input.mPoints[2]);
465  bufferQuat.mPoints[3] = (-input.mPoints[3]);
466  } else {
467  bufferQuat = input;
468  }
469 
470  if (Math.abs(cosHalftheta) >= 1.0) {
471  output.mPoints[0] = (this.mPoints[0]);
472  output.mPoints[1] = (this.mPoints[1]);
473  output.mPoints[2] = (this.mPoints[2]);
474  output.mPoints[3] = (this.mPoints[3]);
475  } else {
476  double sinHalfTheta = Math.sqrt(1.0 - cosHalftheta * cosHalftheta);
477  double halfTheta = Math.acos(cosHalftheta);
478 
479  double ratioA = Math.sin((1 - t) * halfTheta) / sinHalfTheta;
480  double ratioB = Math.sin(t * halfTheta) / sinHalfTheta;
481 
482  //Calculate Quaternion
483  output.mPoints[3] = ((float) (mPoints[3] * ratioA + bufferQuat.mPoints[3] * ratioB));
484  output.mPoints[0] = ((float) (this.mPoints[0] * ratioA + bufferQuat.mPoints[0] * ratioB));
485  output.mPoints[1] = ((float) (this.mPoints[1] * ratioA + bufferQuat.mPoints[1] * ratioB));
486  output.mPoints[2] = ((float) (this.mPoints[2] * ratioA + bufferQuat.mPoints[2] * ratioB));
487  }
488  }

490  /**
491  * Rotates a 3-D vector by this <code>Quaternion</code>.
492  *
493  * @param v the {@link Vector3f} to be rotated.
494  * @return the rotated {@link Vector3f}.
495  */
496  public Vector3f rotateVector(Vector3f v) {
497  float q0 = this.mPoints[3];
498  float q1 = this.mPoints[0];
499  float q2 = this.mPoints[1];
500  float q3 = this.mPoints[2];
501  float v1 = v.getX();
502  float v2 = v.getY();
503  float v3 = v.getZ();
504  float x = (1-2*q2*q2-2*q3*q3)*v1 + 2*(q1*q2+q0*q3)*v2 + 2*(q1*q3-q0*q2)*v3;
505  float y = 2*(q1*q2-q0*q3)*v1 + (1-2*q1*q1-2*q3*q3)*v2 + 2*(q2*q3+q0*q1)*v3;
506  float z = 2*(q1*q3-q0*q2) + (q2*q3-q0*q1)*v2 + (1-2*q1*q1-2*q2*q2)*v3;
507  return new Vector3f(x,y,z);
508  }
