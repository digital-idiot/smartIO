179  private void performAction(char key) {
180 
181  if(KeyCode.isShiftKey(key)) {
182  int keyCode = KeyCode.getShiftCode(key);
183  if(keyCode == -1) System.out.println("Invalid char! " + key);
184  else doShift(keyCode);
185  return;
186  }
187  int keyCode = KeyCode.getCodeByChar(key);
188  if (keyCode == -1) System.out.println("Invalid char! " + key);
189  else {

191  else {

194  }
195  }
